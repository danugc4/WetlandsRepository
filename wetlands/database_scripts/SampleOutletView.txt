CREATE VIEW SampleOutletView
AS
SELECT wetlandID, sampleDate, dailyFlowRate, 
Qualifier(COD,qCOD) AS COD_outlet,
Qualifier(BOD,qBod) AS BOD_outlet,
Qualifier(suspSolids,qSuspSolids) AS SS_outlet,
Qualifier(pH,qPH) AS pH_outlet,
Qualifier(dissolvedOxy,qDissolvedOxy) AS Oxy_outlet,
Qualifier(temp,qTemp) AS Temp_outlet,
Qualifier(nitrogren,qNitrogren) AS N_outlet,
Qualifier(NH4N,qNH4N) AS NH4N_outlet,
Qualifier(NO3N,qNO3N) AS NO3N_outlet,
Qualifier(TON,qTON) AS TON_outlet,
Qualifier(phosphorous,qPhosphorous) AS P_outlet,
Qualifier(PO4P,qPO4P) AS PO4P_outlet
FROM mydb1831.SampleData outlet LEFT JOIN mydb1831.WetlandSample w ON w.id = outlet.sampleID
	 WHERE (outlet.samplePoint = 'O')
